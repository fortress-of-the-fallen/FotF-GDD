\section{Phân tích hệ thống}
\label{sec:analysis}

Chương này phân tích hệ thống ở mức logic cho phạm vi đã triển khai thực tế, nhằm chuyển hoá yêu cầu ở Chương~\ref{sec:requirements} thành mô tả vận hành rõ ràng cho thiết kế và kiểm thử tích hợp.

% ===============================================================
\subsection{Tổng quan phân hệ và ranh giới trách nhiệm}
\label{subsec:analysis_overview}

Hệ thống vận hành theo mô hình client--server:
\begin{itemize}
    \item \textbf{Client}: hiển thị UI, thu thập input, gửi request và áp dụng state trả về.
    \item \textbf{Server}: xác thực, quản lý phiên, xử lý nghiệp vụ và chốt state authoritative.
\end{itemize}

Các phân hệ đang có runtime/API trong baseline:
\begin{itemize}
    \item Auth/\allowbreak Session.
    \item Player State Snapshot (\texttt{/me}).
    \item Quest Sync/\allowbreak Claim.
    \item Battle Session Start/\allowbreak Finish.
    \item Tutorial Progress Sync.
    \item Hero/\allowbreak Skill Progression.
    \item Castle Progression.
\end{itemize}

\subsubsection{UC-01 -- Đăng nhập và thiết lập phiên}
\label{subsubsec:uc-01}

\textbf{Mục tiêu:} xác thực người chơi và cấp token hợp lệ để truy cập API.

\textbf{Tác nhân chính:} Player.

\textbf{Tiền điều kiện:} tài khoản tồn tại và không bị khoá.

\textbf{Hậu điều kiện:} client có access/\allowbreak refresh token; có thể gọi \texttt{/me}.

\textbf{Luồng chính:}
\begin{enumerate}
    \item Player nhập thông tin đăng nhập.
    \item Client gửi request login.
    \item Server xác thực và trả token.
    \item Client lưu token và chuyển sang đồng bộ state.
\end{enumerate}

\textbf{Ngoại lệ:}
\begin{itemize}
    \item Sai thông tin đăng nhập $\rightarrow$ trả lỗi xác thực.
\end{itemize}

\subsubsection{UC-02 -- Đồng bộ player state}
\label{subsubsec:uc-02}

\textbf{Mục tiêu:} nạp trạng thái người chơi nhất quán sau khi xác thực.

\textbf{Tác nhân chính:} Game Client.

\textbf{Tiền điều kiện:} access token hợp lệ.

\textbf{Hậu điều kiện:} state local phản ánh đúng snapshot backend.

\textbf{Luồng chính:}
\begin{enumerate}
    \item Client gọi \texttt{GET /me}.
    \item Server trả profile, statistics, tutorialProgress, currency, inventory, formation, quests.
    \item Client apply state theo thứ tự ưu tiên backend response.
\end{enumerate}

\subsubsection{UC-03 -- Đồng bộ và claim quest}
\label{subsubsec:uc-03}

\textbf{Mục tiêu:} cập nhật tiến độ quest và nhận thưởng theo điều kiện backend.

\textbf{Tác nhân chính:} Player.

\textbf{Tiền điều kiện:} đã có snapshot state, quest tồn tại.

\textbf{Hậu điều kiện:} progress/\allowbreak rewardClaimed được cập nhật nhất quán.

\textbf{Luồng chính:}
\begin{enumerate}
    \item Client gọi endpoint lấy quest.
    \item Player thực hiện claim khi đủ điều kiện.
    \item Server validate, cập nhật dữ liệu và trả trạng thái mới.
    \item Client cập nhật UI quest theo payload trả về.
\end{enumerate}

\subsubsection{UC-04 -- Khởi tạo battle session}
\label{subsubsec:uc-04}

\textbf{Mục tiêu:} tạo phiên battle hợp lệ trước khi vào trận.

\textbf{Tác nhân chính:} Player.

\textbf{Tiền điều kiện:} đã đăng nhập, có formation hợp lệ.

\textbf{Hậu điều kiện:} nhận \texttt{battleId} và thông tin phiên để bắt đầu gameplay.

\textbf{Luồng chính:}
\begin{enumerate}
    \item Client gửi request \texttt{battles/start} với mode, formationName, configVersion.
    \item Server tạo battle session và trả dữ liệu phiên.
    \item Client lưu \texttt{battleId} để dùng cho bước finish.
\end{enumerate}

\subsubsection{UC-05 -- Kết thúc battle và đồng bộ hậu trận}
\label{subsubsec:uc-05}

\textbf{Mục tiêu:} chốt kết quả trận đấu theo backend source-of-truth.

\textbf{Tác nhân chính:} Player.

\textbf{Tiền điều kiện:} có \texttt{battleId} còn hiệu lực.

\textbf{Hậu điều kiện:} reward, statistics, currency, tutorialProgress, questUpdates được cập nhật.

\textbf{Luồng chính:}
\begin{enumerate}
    \item Client gửi \texttt{battles/\{battleId\}/finish} kèm \texttt{idempotencyKey}.
    \item Server validate phiên và chốt kết quả.
    \item Server trả payload hậu trận.
    \item Client cập nhật state và UI theo payload này.
\end{enumerate}

\textbf{Ngoại lệ:}
\begin{itemize}
    \item Retry cùng idempotency key $\rightarrow$ backend không cấp trùng kết quả.
\end{itemize}

\subsubsection{UC-06 -- Đồng bộ tutorial progress}
\label{subsubsec:uc-06}

\textbf{Mục tiêu:} đồng bộ các cờ tutorial liên phiên.

\textbf{Tác nhân chính:} Game Client.

\textbf{Tiền điều kiện:} có thay đổi cờ tutorial từ gameplay.

\textbf{Hậu điều kiện:} backend lưu và trả tutorialProgress mới.

\textbf{Luồng chính:}
\begin{enumerate}
    \item Client gửi \texttt{PATCH /me/tutorial-progress} với các trường cập nhật.
    \item Server ghi nhận và trả trạng thái mới.
    \item Client apply lại cờ tutorial theo response.
\end{enumerate}

\subsubsection{UC-07 -- Nâng cấp hero và skill}
\label{subsubsec:uc-07}

\textbf{Mục tiêu:} cập nhật progression hero/\allowbreak skill theo rule backend.

\textbf{Tác nhân chính:} Player.

\textbf{Tiền điều kiện:} đủ tài nguyên và điều kiện nâng cấp.

\textbf{Hậu điều kiện:} dữ liệu hero/\allowbreak skill/\allowbreak currency được cập nhật đồng bộ.

\textbf{Luồng chính:}
\begin{enumerate}
    \item Player thao tác nâng cấp trên UI Units hoặc UI kỹ năng.
    \item Client gọi endpoint nâng cấp tương ứng.
    \item Server validate điều kiện và trả trạng thái mới.
    \item Client cập nhật lại dữ liệu hiển thị.
\end{enumerate}

\subsubsection{UC-08 -- Nâng cấp castle}
\label{subsubsec:uc-08}

\textbf{Mục tiêu:} thực hiện nâng cấp castle bằng backend contract riêng.

\textbf{Tác nhân chính:} Player.

\textbf{Tiền điều kiện:} đủ tài nguyên và điều kiện nâng cấp castle.

\textbf{Hậu điều kiện:} castle progression và currency mới được trả về cho client.

\textbf{Luồng chính:}
\begin{enumerate}
    \item Player bấm nâng cấp castle.
    \item Client gọi endpoint nâng cấp castle.
    \item Server validate và ghi nhận nâng cấp.
    \item Client cập nhật UI theo response backend.
\end{enumerate}

\subsection{Quy tắc nghiệp vụ và ràng buộc hệ thống}
\label{subsec:analysis_rules}

\subsubsection{Server-authoritative state}
\label{subsec:analysis_rules_authority}
\begin{itemize}
    \item Reward, currency, statistics, quest progress, tutorial flags, castle progression chỉ lấy từ payload backend.
    \item Client không tự suy diễn kết quả hậu trận hoặc kết quả claim.
\end{itemize}

\subsubsection{Idempotency và retry}
\label{subsec:analysis_rules_idempotency}
\begin{itemize}
    \item Các thao tác có khả năng retry phải đi kèm idempotency key.
    \item Backend phải chống cấp trùng dữ liệu khi nhận lại request cùng khoá.
\end{itemize}

\subsubsection{Nhất quán bất đồng bộ}
\label{subsec:analysis_rules_consistency}
\begin{itemize}
    \item Khi có nhiều response gần nhau, client chỉ áp dụng trạng thái mới nhất theo mốc thời gian/version.
    \item Luồng hậu battle phải apply state trước khi mở các UI kế tiếp.
\end{itemize}

\subsubsection{Tương thích cấu hình}
\label{subsec:analysis_rules_config}
\begin{itemize}
    \item Battle session phải gắn với \texttt{configVersion} hợp lệ.
    \item Client và backend cần cùng phiên bản config để tránh lệch tính toán.
\end{itemize}

\subsection{Phân tích dữ liệu ở mức logic}
\label{subsec:analysis_data}

\subsubsection{Dữ liệu tiến trình}
\label{subsec:analysis_data_types}
\begin{itemize}
    \item Auth/session tokens.
    \item Player profile và statistics.
    \item Tutorial progress flags.
    \item Currency và inventory (heroes/\allowbreak skills).
    \item Formation state.
    \item Quest tracks và progress rewards.
    \item Battle session state và dữ liệu hậu trận.
    \item Castle progression state.
\end{itemize}

\subsubsection{Thực thể lõi theo luồng hiện tại}
\label{subsec:analysis_entities}
\begin{itemize}
    \item \textbf{Identity}: account, session/token state.
    \item \textbf{Player snapshot}: profile, statistics, tutorial, currency, inventory, formation, quests.
    \item \textbf{Battle}: phiên start/finish và reward/result state.
    \item \textbf{Progression}: hero/skill upgrade state, castle state.
\end{itemize}

\subsubsection{Quan hệ dữ liệu quan trọng}
\label{subsec:analysis_relationships}
\begin{itemize}
    \item Một account gắn với một định danh player dùng xuyên suốt các luồng auth, state, battle, quest.
    \item Snapshot \texttt{/me} là điểm hợp nhất dữ liệu cho nhiều phân hệ để client đồng bộ nhất quán.
    \item Battle finish, quest claim, tutorial patch và upgrade đều tạo ra thay đổi progression cần phản ánh lại về snapshot hiển thị.
\end{itemize}

\subsection{Tổng kết chương}
\label{subsec:analysis_summary}

Chương này đã phân tích hệ thống theo đúng phạm vi implemented-only: xác thực phiên, đồng bộ trạng thái người chơi, quest, battle session, tutorial sync, nâng cấp hero/\allowbreak skill và nâng cấp castle. Kết quả phân tích là đầu vào trực tiếp cho quyết định thiết kế ở Chương~7 và bám chặt baseline vận hành thực tế.
